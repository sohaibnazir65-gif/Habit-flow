function isScheduled(h, date) {
  const created = (h.createdAt||"").split("T")[0];
  if (created && date < created) return false;
  if (h.frequency==="daily") return true;
  if (h.frequency==="specific") return (h.days||[]).includes(new Date(date+"T00:00:00").getDay());
  return true;
}
function calcStreak(habit) {
  let cur=0,best=0,streak=0,checking=true;
  let d=new Date();
  for(let i=0;i<365&&checking;i++){
    const ds=d.toISOString().split("T")[0];
    if(!isScheduled(habit,ds)){d.setDate(d.getDate()-1);continue;}
    if(habit.completions[ds]?.done){cur++;d.setDate(d.getDate()-1);}
    else if(i===0){d.setDate(d.getDate()-1);}
    else checking=false;
  }
  const sorted=Object.keys(habit.completions).sort();
  if(sorted.length){
    const start=new Date(sorted[0]+"T00:00:00");
    for(let dd=new Date(start);dd<=new Date();dd.setDate(dd.getDate()+1)){
      const ds=dd.toISOString().split("T")[0];
      if(!isScheduled(habit,ds))continue;
      if(habit.completions[ds]?.done){streak++;if(streak>best)best=streak;}
      else streak=0;
    }
  }
  return {cur,best};
}
function getRate(habit,days){
  const sc=days.filter(d=>isScheduled(habit,d));
  if(!sc.length)return 0;
  return Math.round((sc.filter(d=>habit.completions[d]?.done).length/sc.length)*100);
}

function detectCombo(habits,date){
  const doneCats=new Set(habits.filter(h=>isScheduled(h,date)&&h.completions[date]?.done).map(h=>h.category));
  for(const combo of COMBOS){
    if(combo.cats.every(c=>doneCats.has(c))) return combo;
  }
  return null;
}

// ─── Weekly Recap ─────────────────────────────────────────────────────────────
function getWeekRange(){
  const now=new Date();
  const dow=now.getDay();
  const start=new Date(now);start.setDate(now.getDate()-dow);
  const end=new Date(start);end.setDate(start.getDate()+6);
  return{start:start.toISOString().split("T")[0],end:end.toISOString().split("T")[0]};
}
function getWeeklyRecap(habits){
  const days=last7();
  let totalScheduled=0,totalDone=0;
  const habitRates=habits.map(h=>{
    const sched=days.filter(d=>isScheduled(h,d));
    const done=sched.filter(d=>h.completions[d]?.done).length;
    totalScheduled+=sched.length;totalDone+=done;
    return{h,rate:sched.length?Math.round((done/sched.length)*100):0,done,sched:sched.length};
  });
  const overallRate=totalScheduled?Math.round((totalDone/totalScheduled)*100):0;
  const best=[...habitRates].sort((a,b)=>b.rate-a.rate)[0];
  const worst=[...habitRates].sort((a,b)=>a.rate-b.rate)[0];
  const perfectDays=days.filter(d=>{const sc=habits.filter(h=>isScheduled(h,d));return sc.length>0&&sc.every(h=>h.completions[d]?.done);}).length;
  let moodCount=0,topMoodVal=null,moodTally={};
  days.forEach(d=>habits.forEach(h=>{const c=h.completions[d];if(c?.mood){moodCount++;moodTally[c.mood]=(moodTally[c.mood]||0)+1;}}));
  if(Object.keys(moodTally).length) topMoodVal=Object.entries(moodTally).sort((a,b)=>b[1]-a[1])[0][0];
  return{overallRate,best,worst,perfectDays,totalDone,topMoodVal,habitRates};
}

function computeGlobalStats(habits,extra={}){
  let total=0,moodLogs=0,notes=0,perfectDays=0,earlyBird=false;
  habits.forEach(h=>{
    Object.entries(h.completions).forEach(([date,c])=>{
      if(c.done){total++;if(c.completedAt){const hr=new Date(c.completedAt).getHours();if(hr<8)earlyBird=true;}}
      if(c.mood)moodLogs++;
      if(c.note&&c.note.trim())notes++;
    });
  });
  last90().forEach(d=>{
    const dh=habits.filter(x=>isScheduled(x,d));
    if(dh.length>0&&dh.every(x=>x.completions[d]?.done))perfectDays++;
  });
  return{total,moodLogs,notes,perfectDays,earlyBird,pomodoroSessions:extra.pomodoroSessions||0};
}

// ─── Voice Command Hook ───────────────────────────────────────────────────────
function useVoice(habits,onToggle,show,lang){
  const[listening,setListening]=useState(false);
  const[transcript,setTranscript]=useState("");
  const recRef=useRef(null);
  const habitsRef=useRef(habits);
  const toggleRef=useRef(onToggle);
  const showRef=useRef(show);
  const langRef=useRef(lang);
  habitsRef.current=habits; toggleRef.current=onToggle; showRef.current=show; langRef.current=lang;

  const matchHabit=(text)=>{
    const lower=text.toLowerCase();
    let best=null,bestScore=0;
    habitsRef.current.forEach(h=>{
      const name=h.name.toLowerCase();
      const words=name.split(" ");
      let score=0;
      if(lower.includes(name)) score=100;
      else words.forEach(w=>{if(w.length>2&&lower.includes(w))score+=20;});
      if(score>bestScore){bestScore=score;best=h;}
    });
    return bestScore>0?best:null;
  };

  const start=()=>{
    const SR=window.SpeechRecognition||window.webkitSpeechRecognition;
    if(!SR){ showRef.current(langRef.current==="ur"?"Voice support nahi hai is browser mein":"Voice not supported in this browser","error"); return; }
    const rec=new SR();
    rec.lang=langRef.current==="ur"?"ur-PK":"en-US";
    rec.interimResults=false;
    rec.maxAlternatives=1;
    rec.onstart=()=>setListening(true);
    rec.onerror=()=>{setListening(false); showRef.current(langRef.current==="ur"?"Awaaz samajh nahi aayi":"Couldn't understand, try again","error");};
    rec.onend=()=>setListening(false);
    rec.onresult=(e)=>{
      const text=e.results[0][0].transcript;
      setTranscript(text);
      const habit=matchHabit(text);
      if(habit){
        toggleRef.current(habit.id,todayStr());
        showRef.current(`🎤 ${habit.emoji} ${habit.name} ${langRef.current==="ur"?"mukammal!":"completed!"}`,"success");
      } else {
        showRef.current(langRef.current==="ur"?`"${text}" se koi habit match nahi hui`:`No habit matched "${text}"`,"error");
      }
    };
    rec.start();
    recRef.current=rec;
  };
  const stop=()=>{ recRef.current?.stop(); setListening(false); };
  const toggleListen=()=>{ listening?stop():start(); };
  return{listening,transcript,start,stop,toggleListen};
}

// ─── Calendar Screen ──────────────────────────────────────────────────────────
function CalendarScreen({habits}){
  const now=new Date();
  const[year,setYear]=useState(now.getFullYear());
  const[month,setMonth]=useState(now.getMonth());
  const[selected,setSelected]=useState(todayStr());

  const daysInMonth=getDaysInMonth(year,month);
  const firstDay=getFirstDayOfMonth(year,month);
  const today=todayStr();

  const getDayStr=(day)=>`${year}-${String(month+1).padStart(2,"0")}-${String(day).padStart(2,"0")}`;
  const getDayStats=(ds)=>{
    const h=habits.filter(x=>isScheduled(x,ds));
    const done=h.filter(x=>x.completions[ds]?.done).length;
    return{total:h.length,done,pct:h.length>0?Math.round((done/h.length)*100):0};
  };

  const selStats=getDayStats(selected);
  const selHabits=habits.filter(h=>isScheduled(h,selected));

  const prevMonth=()=>{if(month===0){setMonth(11);setYear(y=>y-1);}else setMonth(m=>m-1);};

      <h1 style={{fontFamily:"'Space Grotesk',sans-serif",fontSize:"1.75rem",fontWeight:700,marginBottom:20}}>Calendar 📅</h1>

// ─── Pomodoro Timer ───────────────────────────────────────────────────────────
function PomodoroModal({onClose,settings,onSessionComplete}){
  const WORK=(settings?.work||25)*60;
  const REST=(settings?.rest||5)*60;
  const[phase,setPhase]=useState("work");
  const[sec,setSec]=useState(WORK);
  const[running,setRunning]=useState(false);
  const[sessions,setSessions]=useState(0);
  const[task,setTask]=useState("");
  const ref=useRef(null);

  useEffect(()=>{
    if(running){
      ref.current=setInterval(()=>{
        setSec(s=>{
          if(s<=1){
            clearInterval(ref.current);setRunning(false);
            if(phase==="work"){setSessions(n=>{const nn=n+1;onSessionComplete(nn);return nn;});setPhase("rest");setSec(REST);}
            else{setPhase("work");setSec(WORK);}
            return 0;
          }
          return s-1;
        });
      },1000);
    }
    return()=>clearInterval(ref.current);
  },[running,phase]);

  const mm=String(Math.floor(sec/60)).padStart(2,"0");
  const ss2=String(sec%60).padStart(2,"0");
  const total=phase==="work"?WORK:REST;
  const pct=Math.round(((total-sec)/total)*100);
  const col=phase==="work"?"#6C63FF":"#34D399";

  return(
    <div style={{position:"fixed",inset:0,background:"rgba(0,0,0,0.85)",backdropFilter:"blur(10px)",zIndex:300,display:"flex",alignItems:"center",justifyContent:"center",padding:20}}
      onClick={e=>e.target===e.currentTarget&&onClose()}>
      <div style={{...S.card,padding:28,maxWidth:340,width:"100%",borderRadius:24,textAlign:"center"}}>
        <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:20}}>
          <span style={{fontFamily:"'Space Grotesk',sans-serif",fontWeight:700,fontSize:"1.1rem"}}>🍅 Pomodoro</span>
          <button onClick={onClose} style={{...S.btn,background:"var(--bg3)",border:"1px solid var(--bdr)",borderRadius:8,padding:"6px 10px",color:"var(--t1)"}}>✕</button>
        </div>
        {/* Task input */}
        <input value={task} onChange={e=>setTask(e.target.value)} placeholder="What are you working on? (optional)"
          style={{...S.inp,marginBottom:16,textAlign:"center",fontSize:"0.875rem"}}
          onFocus={e=>e.target.style.borderColor="#6C63FF"} onBlur={e=>e.target.style.borderColor="var(--bdr)"}/>
        {/* Ring */}
        <div style={{display:"flex",justifyContent:"center",marginBottom:20}}>
          <Ring pct={pct} color={col} size={160} stroke={10}>
            <div>
              <div style={{fontFamily:"'Space Grotesk',sans-serif",fontSize:"2.6rem",fontWeight:700,color:col,lineHeight:1}}>{mm}:{ss2}</div>
              <div style={{fontSize:"0.78rem",color:"var(--t2)",fontWeight:600,textTransform:"uppercase",marginTop:4}}>{phase==="work"?"Focus Time":"Break Time"}</div>
            </div>
          </Ring>
        </div>
        {/* Phase buttons */}
        <div style={{display:"flex",gap:8,marginBottom:16,justifyContent:"center"}}>
          {[["work","🧠 Focus"],["rest","☕ Break"]].map(([p,l])=>(
            <button key={p} onClick={()=>{setPhase(p);setSec(p==="work"?WORK:REST);setRunning(false);}} style={{...S.btn,padding:"8px 18px",borderRadius:10,background:phase===p?col:"var(--bg3)",color:phase===p?"white":"var(--t2)",border:`1px solid ${phase===p?col:"var(--bdr)"}`,fontSize:"0.85rem",fontWeight:700}}>{l}</button>
          ))}
        </div>
        {/* Controls */}
        <div style={{display:"flex",gap:10,justifyContent:"center",marginBottom:18}}>
          <button onClick={()=>setRunning(r=>!r)} style={{...S.btn,flex:1,padding:"14px",borderRadius:12,background:col,color:"white",fontSize:"1rem",fontWeight:700}}>
            {running?"⏸ Pause":"▶ Start"}
          </button>
          <button onClick={()=>{setRunning(false);setSec(phase==="work"?WORK:REST);}} style={{...S.btn,padding:"14px 18px",borderRadius:12,background:"var(--bg3)",border:"1px solid var(--bdr)",color:"var(--t2)",fontSize:"1rem"}}>↺</button>
        </div>
        <div style={{fontSize:"0.78rem",color:"var(--t2)"}}>Sessions today: <strong style={{color:"var(--t1)"}}>{sessions}</strong></div>
        <style>{`@keyframes spin{to{transform:rotate(360deg)}}`}</style>
      </div>
    </div>
  );
}

// ─── Habit DNA (radar chart) ──────────────────────────────────────────────────
function HabitDNA({habits}){
  const cats=CATS.map(cat=>{
    const h=habits.filter(x=>x.category===cat);
    if(!h.length)return null;
    const avgRate=Math.round(h.reduce((a,x)=>a+getRate(x,last30()),0)/h.length);
    return{cat,score:avgRate};
  }).filter(Boolean);

  if(cats.length<3) return null;

  const size=220,cx=size/2,cy=size/2,maxR=80;
  const n=cats.length;
  const angle=(i)=>(Math.PI*2*i)/n-Math.PI/2;
  const pointAt=(i,r)=>({x:cx+r*Math.cos(angle(i)),y:cy+r*Math.sin(angle(i))});
  const dataPoints=cats.map((c,i)=>pointAt(i,(c.score/100)*maxR));
  const pathD=dataPoints.map((p,i)=>`${i===0?"M":"L"}${p.x},${p.y}`).join(" ")+"Z";
  const rings=[0.25,0.5,0.75,1];

  return(
    <div style={{display:"flex",flexDirection:"column",alignItems:"center"}}>
      <svg width={size} height={size}>
        {/* Grid rings */}
        {rings.map((r,ri)=>{
          const ringPts=cats.map((_,i)=>pointAt(i,r*maxR));
          const d=ringPts.map((p,i)=>`${i===0?"M":"L"}${p.x},${p.y}`).join(" ")+"Z";
          return <path key={ri} d={d} fill="none" stroke="var(--bdr)" strokeWidth={1}/>;
        })}
        {/* Spokes */}
        {cats.map((_,i)=>{const p=pointAt(i,maxR);return<line key={i} x1={cx} y1={cy} x2={p.x} y2={p.y} stroke="var(--bdr)" strokeWidth={1}/>;})}
        {/* Data area */}
        <path d={pathD} fill="#6C63FF33" stroke="#6C63FF" strokeWidth={2.5}/>
        {dataPoints.map((p,i)=><circle key={i} cx={p.x} cy={p.y} r={4} fill="#6C63FF"/>)}
        {/* Labels */}
        {cats.map((c,i)=>{
          const p=pointAt(i,maxR+22);
          return <text key={i} x={p.x} y={p.y} fontSize={10} fontWeight={700} fill="var(--t2)" textAnchor="middle" dominantBaseline="middle">{c.cat}</text>;
        })}
      </svg>
      <div style={{display:"flex",gap:14,flexWrap:"wrap",justifyContent:"center",marginTop:8}}>
        {cats.map(c=>(
          <div key={c.cat} style={{fontSize:"0.7rem",color:"var(--t2)",display:"flex",alignItems:"center",gap:4}}>
            <span style={{width:8,height:8,borderRadius:2,background:"#6C63FF"}}/>{c.cat}: <strong style={{color:"var(--t1)"}}>{c.score}%</strong>
          </div>
        ))}
      </div>
    </div>
  );
}

  const bestStreak=Math.max(0,...habits.map(h=>calcStreak(h).best));

  const exportCSV=()=>{const rows=["Habit,Category,Date,Done,Mood,Energy,Note"];habits.forEach(h=>Object.entries(h.completions).forEach(([d,c])=>rows.push(`"${h.name}",${h.category},${d},${c.done?"Yes":"No"},${c.mood||""},${c.energy||""},"${(c.note||"").replace(/"/g,'""')}"`))); const b=new Blob([rows.join("\n")],{type:"text/csv"});const a=document.createElement("a");a.href=URL.createObjectURL(b);a.download=`habitflow-${todayStr()}.csv`;a.click();showT("✓ CSV exported!");};
  const handleImp=async e=>{const f=e.target.files?.[0];if(!f)return;const r=new FileReader();r.onload=ev=>{try{const d=JSON.parse(ev.target.result);onImport(d.habits||d);showT("✓ Imported!");}catch{showT("✗ Invalid file");}};r.readAsText(f);e.target.value="";};

  const sections=[
    {s:"Appearance",items:[
      {icon:"🌓",label:t.theme,desc:theme==="dark"?t.currentlyDark:t.currentlyLight,action:onToggleTheme,right:<div style={{width:46,height:26,borderRadius:999,background:theme==="dark"?"#6C63FF":"var(--bg4)",position:"relative",transition:"background 0.3s"}}><div style={{width:22,height:22,borderRadius:"50%",background:"white",position:"absolute",top:2,left:theme==="dark"?22:2,transition:"left 0.3s",boxShadow:"0 1px 3px rgba(0,0,0,0.3)"}}/></div>},
      {icon:"🌐",label:t.language,desc:lang==="ur"?"Roman Urdu":"English",action:onToggleLang,right:<span style={{fontSize:"0.78rem",fontWeight:700,color:"#6C63FF",background:"#6C63FF18",borderRadius:999,padding:"4px 10px"}}>{lang==="ur"?"اردو":"EN"}</span>},

    {s:"Backup",items:[{icon:"⬇️",label:t.exportJSON,desc:"Full backup with all history",action:exportJSON},{icon:"📊",label:t.exportCSV,desc:"Spreadsheet-friendly",action:exportCSV},{icon:"⬆️",label:t.importJSON,desc:"Restore from backup",action:()=>fileRef.current?.click()}]},
    {s:"Danger Zone",items:[{icon:"🗑️",label:t.deleteAll,desc:t.cannotUndo,action:()=>{if(confirm("Delete everything permanently?"))onDeleteAll();},danger:true}]},
  ];

  return(
    <div style={{padding:"24px 16px 100px"}}>
      <h1 style={{fontFamily:"'Space Grotesk',sans-serif",fontSize:"1.75rem",fontWeight:700,marginBottom:20}}>Settings</h1>
      <div style={{background:"linear-gradient(135deg,#6C63FF18,#A78BFA18)",border:"1px solid #6C63FF33",borderRadius:20,padding:24,marginBottom:24,textAlign:"center"}}>
        <div style={{fontSize:"2.5rem",marginBottom:8}}>💫</div>
        <div style={{fontFamily:"'Space Grotesk',sans-serif",fontSize:"1.2rem",fontWeight:700}}>HabitFlow</div>
        <div style={{color:"var(--t2)",fontSize:"0.8rem",marginTop:4,marginBottom:16}}>Build better habits, one day at a time</div>
        <div style={{display:"flex",justifyContent:"center",gap:28}}>
          {[{v:habits.filter(h=>!h.archived).length,l:"Active"},{v:totalDone,l:"Done"},{v:bestStreak,l:"Best Streak"}].map(s=>(<div key={s.l}><div style={{fontFamily:"'Space Grotesk',sans-serif",fontWeight:700,fontSize:"1.5rem",color:"#6C63FF"}}>{s.v}</div><div style={{fontSize:"0.7rem",color:"var(--t2)",fontWeight:600}}>{s.l}</div></div>))}
        </div>
      </div>
      {sections.map(({s,items})=>(
        <div key={s} style={{marginBottom:20}}>
          <div style={{fontSize:"0.7rem",fontWeight:700,color:"var(--t3)",textTransform:"uppercase",letterSpacing:"0.08em",marginBottom:8,paddingLeft:4}}>{s}</div>
          <div style={{...S.card,borderRadius:16,overflow:"hidden"}}>
            {items.map((item,i)=>(<button key={item.label} onClick={item.action} style={{...S.btn,display:"flex",alignItems:"center",gap:14,width:"100%",padding:"16px",background:"none",borderBottom:i<items.length-1?"1px solid var(--bdr)":"none",transition:"background 0.15s"}} onMouseEnter={e=>e.currentTarget.style.background="var(--bg3)"} onMouseLeave={e=>e.currentTarget.style.background="none"}><span style={{fontSize:"1.1rem",flexShrink:0}}>{item.icon}</span><div style={{flex:1,textAlign:"left"}}><div style={{fontWeight:600,fontSize:"0.9375rem",color:item.danger?"#F87171":"var(--t1)"}}>{item.label}</div><div style={{fontSize:"0.78rem",color:"var(--t2)",marginTop:1}}>{item.desc}</div></div>{item.right||<span style={{color:"var(--t3)",fontSize:"1rem"}}>›</span>}</button>))}

  const[editing,setEditing]=useState(null);

  useEffect(()=>{document.documentElement.setAttribute("data-theme",state.theme);},[state.theme]);

    const active=state.habits.filter(h=>!h.archived);
    const stats={...computeGlobalStats(active),pomodoroSessions:state.pomodoroSessions||0};
    const newB=BADGES.filter(b=>!(state.earnedBadges||[]).includes(b.id)&&b.check(active,stats)).map(b=>b.id);
    if(newB.length>0){
      show("🏆 "+newB.map(id=>BADGES.find(b=>b.id===id)?.icon+" "+BADGES.find(b=>b.id===id)?.name).join(", "),"success");
      setState(s=>({...s,earnedBadges:[...(s.earnedBadges||[]),...newB]}));
    }
  },[state.habits,state.pomodoroSessions]);

  const habits=state.habits.filter(h=>!h.archived);
  const setH=fn=>setState(s=>({...s,habits:fn(s.habits)}));
  const addHabit=d=>{setH(h=>[...h,{...d,id:uid(),createdAt:new Date().toISOString(),completions:{},archived:false}]);show("✨ Habit added!","success");setShowForm(false);};
  const editHabit=d=>{setH(h=>h.map(x=>x.id===editing.id?{...x,...d}:x));show("✓ Updated!","success");setShowForm(false);setEditing(null);};
  const delHabit=id=>{setH(h=>h.filter(x=>x.id!==id));show("Deleted");};
  const archiveH=id=>{setH(h=>h.map(x=>x.id===id?{...x,archived:true}:x));show("📦 Archived");};
  const toggle=(id,date)=>{
    setH(h=>{
      const updated=h.map(x=>{if(x.id!==id)return x;const dn=!x.completions[date]?.done;return{...x,completions:{...x.completions,[date]:{...x.completions[date],date,done:dn,completedAt:dn?new Date().toISOString():undefined}}};});
      const target=updated.find(x=>x.id===id);
      const isDone=target?.completions[date]?.done;
      if(isDone){
        const combo=detectCombo(updated.filter(x=>!x.archived),date);
        if(combo) show(`${combo.icon} ${(state.lang||"en")==="ur"?combo.msgUr:combo.msg}`,"success");
        else show("🔥 Done!","success");
      }
      return updated;
    });
  };
  const voice=useVoice(habits,toggle,show,state.lang||"en");
  const updComp=(id,date,upd)=>{setH(h=>h.map(x=>x.id!==id?x:{...x,completions:{...x.completions,[date]:{...x.completions[date],date,...upd}}}));};
  const updWater=cups=>setState(s=>({...s,water:{...(s.water||{}),[todayStr()]:cups}}));
  const updWeight=(date,val)=>setState(s=>({...s,weightLog:{...(s.weightLog||{}),[date]:val}}));
  const updReflection=(date,text)=>setState(s=>({...s,dailyReflections:{...(s.dailyReflections||{}),[date]:text}}));
  const detailFresh=detail?habits.find(h=>h.id===detail.id)||null:null;

  const t=useT(state.lang||"en");
  const TABS=[{id:"today",icon:"🏠",label:t.today},{id:"calendar",icon:"📅",label:t.calendar},{id:"stats",icon:"📊",label:t.stats},{id:"weight",icon:"⚖️",label:t.weight},{id:"badges",icon:"🏆",label:t.badges},{id:"settings",icon:"⚙️",label:t.more}];
  const cp={onToggle:toggle,onEdit:h=>{setEditing(h);setShowForm(true);},onArchive:archiveH,onDelete:delHabit,onDetail:h=>setDetail(habits.find(x=>x.id===h.id)||h),onAdd:()=>{setEditing(null);setShowForm(true);}};

  return(
    <>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700&family=Space+Grotesk:wght@500;700&display=swap');
        *,*::before,*::after{box-sizing:border-box;margin:0;padding:0}

        [data-theme="dark"]{--bg1:#0F0F1A;--bg2:#161624;--bg3:#1E1E30;--bg4:#252538;--bdr:rgba(255,255,255,0.08);--t1:#F0F0FF;--t2:#9090C0;--t3:#5050A0;--shadow:0 4px 24px rgba(0,0,0,0.4)}
        [data-theme="light"]{--bg1:#F0F0FF;--bg2:#FFFFFF;--bg3:#EBEBFF;--bg4:#DCDCF5;--bdr:rgba(0,0,0,0.08);--t1:#0F0F2A;--t2:#5050A0;--t3:#9090C0;--shadow:0 4px 24px rgba(108,99,255,0.1)}
        body{font-family:'Inter',sans-serif;background:var(--bg1);color:var(--t1);min-height:100dvh;-webkit-font-smoothing:antialiased;transition:background 0.3s,color 0.3s;overscroll-behavior:none}
        #root{min-height:100dvh;display:flex;flex-direction:column}
        textarea,input,select{font-family:inherit;color:var(--t1)}
        ::-webkit-scrollbar{width:3px}::-webkit-scrollbar-track{background:transparent}::-webkit-scrollbar-thumb{background:var(--bg4);border-radius:3px}
        button:focus-visible{outline:2px solid #6C63FF;outline-offset:2px}
        @keyframes spin{to{transform:rotate(360deg)}}
        @media(prefers-reduced-motion:reduce){*{transition-duration:0.01ms!important;animation-duration:0.01ms!important}}
